## Migration SQL

```sql
CREATE SCHEMA "telemetry";
CREATE TYPE "telemetry"."workflow_event_type" AS ENUM(
  'item_state_changed', 'step_completed', 'story_changed', 'gap_found', 'flow_issue'
);
CREATE TABLE "telemetry"."workflow_events" (
  "event_id" uuid PRIMARY KEY DEFAULT gen_random_uuid() NOT NULL,
  "event_type" "telemetry"."workflow_event_type" NOT NULL,
  "event_version" integer DEFAULT 1 NOT NULL,
  "ts" timestamp DEFAULT now() NOT NULL,
  "run_id" text,
  "item_id" text,
  "workflow_name" text,
  "agent_role" text,
  "status" text,
  "payload" jsonb
);
-- Plus 8 indexes (1 unique + 7 query)
```
