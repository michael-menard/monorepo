## Story

**I want** wishlist items displayed as cards,
**so that** I can see item details at a glance.

2. ⬜ Shows name, type badge, date
3. ⬜ External link indicator

5. ⬜ Hover state

## Dev Notes

```typescript
interface WishlistItemCardProps {
  item: WishlistItem
  onEdit: () => void
  onDelete: () => void
}

const typeBadgeStyles = {
  moc: 'bg-blue-100 text-blue-700',
  lego: 'bg-yellow-100 text-yellow-700',
  alt: 'bg-purple-100 text-purple-700',
}

export function WishlistItemCard({ item, onEdit, onDelete }: WishlistItemCardProps) {
  return (
    <Card className="p-4 hover:shadow-md transition-shadow">
      <div className="flex items-start justify-between">
        <div>
          <div className="flex items-center gap-2 mb-1">
            <Badge className={typeBadgeStyles[item.type]}>{item.type.toUpperCase()}</Badge>
            <span className="text-sm text-muted-foreground">
              {formatDate(item.createdAt)}
            </span>
          </div>
          <h3 className="font-medium">{item.name}</h3>
          <a href={item.url} target="_blank" rel="noopener" className="text-sm text-primary flex items-center gap-1">
            Visit link <ExternalLink className="h-3 w-3" />
          </a>
        </div>
        <div className="flex gap-1">
          <Button variant="ghost" size="icon" onClick={onEdit}><Pencil className="h-4 w-4" /></Button>
          <Button variant="ghost" size="icon" onClick={onDelete}><Trash2 className="h-4 w-4" /></Button>
        </div>
      </div>
    </Card>
  )
}
```
