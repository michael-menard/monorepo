## Story

**I want** to see the selected image large,
**so that** I can view MOC details clearly.

## Dev Notes

```typescript
interface MainImageDisplayProps {
  image: MocImage
  onOpenLightbox: () => void
}

export function MainImageDisplay({ image, onOpenLightbox }: MainImageDisplayProps) {
  return (
    <button
      onClick={onOpenLightbox}
      className="w-full cursor-zoom-in"
    >
      <AspectRatio ratio={4/3} className="bg-muted rounded-lg overflow-hidden">
        <img
          src={image.url}
          alt={image.alt || 'MOC Image'}
          className="w-full h-full object-contain"
        />
      </AspectRatio>
    </button>
  )
}
```
