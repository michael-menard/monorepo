## Dev Notes

```typescript
export const Route = createFileRoute('/gallery/$mocId')({
  component: MocDetailPage,
})

function MocDetailPage() {
  const { mocId } = Route.useParams()
  const { data: moc, isLoading, error } = useGetMocByIdQuery(mocId)

  if (isLoading) return <MocDetailSkeleton />
  if (error || !moc) return <NotFoundPage />

  return (
    <div className="container py-6">
      <MocDetailHeader moc={moc} />
      <MocImages images={moc.images} />
      <MocFiles files={moc.files} />
    </div>
  )
}
```
