## Dev Notes

```typescript
getMocs: builder.query<MocListResponse, MocQueryParams>({
  query: (params) => ({
    url: '/mocs',
    params: {
      page: params.page || 1,
      limit: params.limit || 20,
      search: params.search,
      tags: params.tags?.join(','),
      theme: params.theme,
    },
  }),
  providesTags: ['Mocs'],
}),

interface MocListResponse {
  items: MocSummary[]
  total: number
  page: number
  limit: number
}
```
