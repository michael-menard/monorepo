## Dev Notes

```typescript
interface ThemeFilterProps {
  selected: string | null
  onChange: (theme: string | null) => void
  themes: string[]
}

export function ThemeFilter({ selected, onChange, themes }: ThemeFilterProps) {
  return (
    <Select value={selected || ''} onValueChange={(v) => onChange(v || null)}>
      <SelectTrigger>
        <SelectValue placeholder="All Themes" />
      </SelectTrigger>
      <SelectContent>
        <SelectItem value="">All Themes</SelectItem>
        {themes.map((theme) => (
          <SelectItem key={theme} value={theme}>{theme}</SelectItem>
        ))}
      </SelectContent>
    </Select>
  )
}
```
