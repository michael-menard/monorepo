## Dev Notes

```typescript
export const Route = createFileRoute('/instructions/new')({
  component: AddMocPage,
})

function AddMocPage() {
  const navigate = useNavigate()

  const handleSuccess = (moc: MocDetail) => {
    navigate({ to: `/gallery/${moc.id}` })
  }

  return (
    <div className="container py-6 max-w-2xl">
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-bold">Add New MOC</h1>
        <Button variant="ghost" asChild>
          <Link to="/gallery">Cancel</Link>
        </Button>
      </div>
      <MocForm onSuccess={handleSuccess} />
    </div>
  )
}
```
