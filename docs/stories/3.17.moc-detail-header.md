## Dev Notes

```typescript
interface MocDetailHeaderProps {
  moc: MocDetail
}

export function MocDetailHeader({ moc }: MocDetailHeaderProps) {
  return (
    <header className="mb-8">
      <h1 className="text-3xl font-bold mb-2">{moc.title}</h1>
      <div className="flex flex-wrap items-center gap-2 mb-4">
        <Badge variant="secondary">{moc.theme}</Badge>
        <span className="text-muted-foreground">•</span>
        <span className="text-muted-foreground">{moc.pieceCount} pieces</span>
      </div>
      {moc.description && (
        <p className="text-muted-foreground">{moc.description}</p>
      )}
      <div className="flex flex-wrap gap-1 mt-4">
        {moc.tags.map((tag) => (
          <Badge key={tag} variant="outline">{tag}</Badge>
        ))}
      </div>
      <p className="text-sm text-muted-foreground mt-4">
        Added {formatDate(moc.createdAt)}
      </p>
    </header>
  )
}
```
