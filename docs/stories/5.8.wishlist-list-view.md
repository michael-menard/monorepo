## Dev Notes

```typescript
interface WishlistListViewProps {
  items: WishlistItem[]
  onEdit: (item: WishlistItem) => void
  onDelete: (item: WishlistItem) => void
}

export function WishlistListView({ items, onEdit, onDelete }: WishlistListViewProps) {
  if (items.length === 0) {
    return <WishlistEmptyState />
  }

  return (
    <div className="space-y-3">
      {items.map((item) => (
        <WishlistItemCard
          key={item.id}
          item={item}
          onEdit={() => onEdit(item)}
          onDelete={() => onDelete(item)}
        />
      ))}
    </div>
  )
}
```
