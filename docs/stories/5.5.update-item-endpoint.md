updateWishlistItem: builder.mutation<WishlistItem, UpdateWishlistItemInput>({
  query: ({ id, ...data }) => ({
    url: `/wishlist/${id}`,
    method: 'PUT',
    body: data,
  }),
  invalidatesTags: ['WishlistItems'],
}),

interface UpdateWishlistItemInput {
  id: string
  name?: string
  url?: string
  type?: 'moc' | 'lego' | 'alt'
