## Dev Notes

### Component Structure

```typescript
export function QuickActions() {
  const actions = [
    { to: '/instructions/new', icon: Plus, label: 'Add MOC', variant: 'default' },
    { to: '/gallery', icon: Images, label: 'Browse Gallery', variant: 'outline' },
    { to: '/wishlist', icon: Heart, label: 'View Wishlist', variant: 'outline' },
  ]

  return (
    <div className="flex flex-wrap gap-2">
      {actions.map((action) => (
        <Button key={action.to} variant={action.variant} asChild>
          <Link to={action.to}>
            <action.icon className="mr-2 h-4 w-4" />
            {action.label}
          </Link>
        </Button>
      ))}
    </div>
  )
}
```
