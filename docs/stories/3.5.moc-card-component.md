## Story

**I want** MOCs displayed as cards,
**so that** I can browse my collection visually.

## Dev Notes

```typescript
interface MocCardProps {
  moc: MocSummary
}

export function MocCard({ moc }: MocCardProps) {
  return (
    <Link to={`/gallery/${moc.id}`}>
      <Card className="overflow-hidden hover:shadow-lg transition-shadow">
        <AspectRatio ratio={4/3}>
          <img
            src={moc.thumbnail || '/placeholder-moc.png'}
            alt={moc.title}
            className="object-cover w-full h-full"
          />
        </AspectRatio>
        <CardContent className="p-3">
          <h3 className="font-semibold truncate">{moc.title}</h3>
          <p className="text-sm text-muted-foreground">{moc.pieceCount} pieces</p>
          <div className="flex flex-wrap gap-1 mt-2">
            {moc.tags.slice(0, 3).map(tag => (
              <Badge key={tag} variant="secondary" size="sm">{tag}</Badge>
            ))}
          </div>
        </CardContent>
      </Card>
    </Link>
  )
}
```
